# get user input
```python
name = input(<string>, ...)
```


# format string
```python
name = "world"
print(f"hello, {name}")
```

# remove whitespace from str
```python
name = name.strip()
```

# capitalize string
```python
name = name.capitalize()
```

# capitalize each word of a string
```python
name = name.title()
```

# split str
```python
name = name.split()
```

# convert str to int
```python
x = int(x)
```

# format float
```python
print(f"result: {z:,}") # 1000 becomes 1,000
```
